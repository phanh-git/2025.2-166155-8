-- This query updates the stock quantity and Vietnamese units for each site/product pair
INSERT INTO site_merchandises (site_code, merchandise_code, quantity, unit)
VALUES 
    ('JP01', 'P001', 80, 'hộp'),
    ('JP01', 'P002', 100, 'thùng'),
    ('KR01', 'P001', 60, 'hộp'),
    ('KR01', 'P003', 90, 'chai'),
    ('SG01', 'P001', 30, 'hộp'),
    ('SG01', 'P002', 45, 'thùng'),
    ('SG01', 'P003', 20, 'chai')
ON CONFLICT (site_code, merchandise_code) 
DO UPDATE SET 
    quantity = EXCLUDED.quantity, 
    unit = EXCLUDED.unit;

-- Verification query
SELECT sm.site_code, s.name as site_name, sm.merchandise_code, m.name as item_name, sm.quantity, sm.unit
FROM site_merchandises sm
JOIN sites s ON sm.site_code = s.code
JOIN merchandises m ON sm.merchandise_code = m.code
ORDER BY sm.site_code, sm.merchandise_code;
