-- PostgreSQL query to update merchandise data with Vietnamese characters
-- Paste this into pgAdmin 4 Query Tool

INSERT INTO merchandises (code, name, unit)
VALUES 
    ('P001', 'Sữa bột nhập khẩu', 'hộp'),
    ('P002', 'Bánh quy', 'thùng'),
    ('P003', 'Dầu olive', 'chai')
ON CONFLICT (code) 
DO UPDATE SET 
    name = EXCLUDED.name, 
    unit = EXCLUDED.unit;

-- Verification query
SELECT * FROM merchandises WHERE code IN ('P001', 'P002', 'P003') ORDER BY code;
